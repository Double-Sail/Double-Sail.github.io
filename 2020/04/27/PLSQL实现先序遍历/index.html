PLSQL实现先序遍历

理论

传统的遍历方法需要判断是否有左孩子右兄弟，还有递归的操作。那么想到可不可以不进行递归，直接一条线的去进行遍历，这样就可以降低时间复杂度。

首先根节点赋值（1，2），后续插入节点则根据一定的排序操作进行赋值更新，这样每一个节点都会被赋以左值和右值。进行遍历时直接按照左节点的大小进行排序，并输出。

建表

create table tree(
 nodename nvarchar2(50) not null,--节点名
 levelnum number(8) not null,--层号
 leftvalue number(8) not null,--左值
 rightvalue number(8) not null);--右值

alter table tree add constraint pk_nodename primary key (nodename);

表的结构:

SQL>  select * from tree;

NODENAME                                            LEVELNUM LEFTVALUE RIGHTVALUE
--------------------------------------------------  -------- --------- ----------

插入节点与获取子孙节点

包头：

create or replace package tree_package is
  --type retcursor is ref cursor;    
  -- Author  : 86135
  -- Created : 2020/4/12 20:27:16
  -- Purpose : 

   procedure insertnode --插入节点
     (parentname in varchar2,--父节点名，插入最开始的节点时为空
      nodename in varchar2
      );--节点名
   procedure hqzsjd--获取子孙节点
     (parentname in varchar2,--父节点名，插入最开始的节点时为空
      zsjd out sys_refcursor
      );--子孙节点名

end tree_package;

包体：

create or replace package body tree_package is
    --插入节点
   procedure insertnode (parentname in varchar2,nodename in varchar2 ) is
    parentright tree.rightvalue%type; --父节点右值
    parentlevel tree.levelnum%type; --父节点层次
   begin
    if parentname is null then
      insert into tree values (nodename, 1, 1, 2);--即如果是根节点，则直接输入节点名
    else
      --如果输入的父节点名在表中已经存在，则将父节点的右值和层次赋给一个变量，以便于后面运用。
      select rightvalue, levelnum
        into parentright, parentlevel
        from tree
       where nodename = parentname;

      --如果节点的左值大于当前输入节点的父节点的右值名称减一，则左值加2。
      --因为在插入了一个左孩子以后，所有的右兄弟以及右兄弟的父节点的左右值都需要加2，根节点只需要右节点加2即可。
      update tree
         set leftvalue = leftvalue + 2
       where leftvalue > parentright - 1;

       --如果节点的右值大于当前输入节点的父节点的右值名称减一，则右值加2。因为相当于插入了一个节点，先序遍历的最后回归位置是父节点的右值，所以右值加2.
      update tree
         set rightvalue = rightvalue + 2
       where rightvalue > parentright - 1;

       --插入新的节点，左值为父节点的右值，右值为父节点的右值加1.
      insert into tree
      values
        (nodename, parentlevel + 1, parentright, parentright + 1);
     end if;
   end insertnode;

  --输出子孙节点
   procedure hqzsjd --获取子孙节点
  (parentname in varchar2, --父节点名
   zsjd out sys_refcursor--子孙节点名
   )is
    parentleft  tree.leftvalue%type; --父节点左值
    parentright tree.rightvalue%type; --父节点右值
   begin
     --将父节点的左值和右值赋给两个变量，用于后续的范围限定
    select leftvalue,rightvalue
      into parentleft,parentright
      from tree
     where nodename = parentname;

     --直接输出左值在父节点左值和右值之间的子节点。
    open zsjd for
      select nodename
        from tree
       where leftvalue between parentleft and parentright
       order by leftvalue asc;
   end hqzsjd;
end tree_package;
